function doLogin() {
  const u = document.getElementById('login-user').value.trim();
  const p = document.getElementById('login-pass').value;
  const acc = db.accounts.find(a => a.username === u && a.password === p);
  if (!acc) {
    document.getElementById('login-error').textContent = '// AUTHENTICATION FAILED';
    document.getElementById('login-pass').value = '';
    return;
  }
  session = acc;
  localStorage.setItem(SESS_KEY, acc.id);
  document.getElementById('login-screen').style.display = 'none';
  document.getElementById('app').style.display = 'block';
  document.getElementById('topbar-username').textContent = acc.display || acc.username;
  initApp();
}

function doLogout() {
  session = null;
  localStorage.removeItem(SESS_KEY);
  document.getElementById('app').style.display = 'none';
  document.getElementById('login-screen').style.display = 'flex';
  document.getElementById('login-pass').value = '';
  document.getElementById('login-error').textContent = '';
}

function editPost(id) {

  document.getElementById('editor-title-label').textContent = 'Editing: ' + (p.title || 'Post');

  document.getElementById('post-content').value = p.content || '';

  document.getElementById('post-date').value = p.date || todayStr();

    date: document.getElementById('post-date').value,
    status: statusOverride || document.getElementById('post-status').value,
    author: document.getElementById('post-author').value.trim() || session.display,
    slug,
    coverUrl: document.getElementById('post-cover-url').value.trim(),
    coverDataURL: editorCoverDataURL,
    attachments: [...editorAttachments],
    links: [...editorLinks],
    updatedAt: Date.now(),
    createdBy: session.username,
  };

  if (editingPostId) {
    const idx = db.posts.findIndex(p => p.id === editingPostId);
    if (idx >= 0) db.posts[idx] = post;
    logActivity(`Edited post: "${title}" (${post.status})`);
  } else {
    post.createdAt = Date.now();
    db.posts.push(post);
    editingPostId = post.id;
    logActivity(`Created post: "${title}" (${post.status})`);
  }

  saveDB(db);
  refreshStats();
  renderPostList();
  populateExportSelect();
  document.getElementById('editor-title-label').textContent = 'Editing: ' + title;
  notify(`Post ${post.status === 'published' ? 'published' : 'saved as draft'}!`, 'success');
}

// â”€â”€â”€ FILE ATTACHMENTS â”€â”€â”€
function handleFiles(e) {
  Array.from(e.target.files).forEach(file => addAttachment(file));
}

function addAttachment(file) {
  const reader = new FileReader();
  reader.onload = ev => {
    editorAttachments.push({
      id: uid(),
      name: file.name,
      type: file.type,
      size: file.size,
      dataURL: ev.target.result,
    });
    renderAttachmentList();
  };
  reader.readAsDataURL(file);
}

function openAccountModal(id) {
  document.getElementById('modal-account-id').value = id || '';
  if (id) {
    const a = db.accounts.find(x => x.id === id);
    document.getElementById('modal-uname').value = a.username;
    document.getElementById('modal-display').value = a.display || '';
    document.getElementById('modal-pass').value = '';
    document.getElementById('modal-role').value = a.role;
    // Non-superadmin can only edit own account
    document.getElementById('modal-role').disabled = session.role !== 'superadmin';
  } else {
    document.getElementById('modal-uname').value = '';
    document.getElementById('modal-display').value = '';
    document.getElementById('modal-pass').value = '';
    document.getElementById('modal-role').value = 'admin';
    document.getElementById('modal-role').disabled = false;
  }
  document.getElementById('modal-account').classList.add('open');
}

function saveAccount() {
  const id   = document.getElementById('modal-account-id').value;
  const uname = document.getElementById('modal-uname').value.trim();
  const disp  = document.getElementById('modal-display').value.trim();
  const pass  = document.getElementById('modal-pass').value;
  const role  = document.getElementById('modal-role').value;

  if (!uname) { notify('Username required', 'error'); return; }
  if (!id && db.accounts.find(a => a.username === uname)) {
    notify('Username already taken', 'error'); return;
  }

  if (id) {
    const a = db.accounts.find(x => x.id === id);
    a.username = uname;
    a.display  = disp;
    if (pass) a.password = pass;
    if (session.role === 'superadmin') a.role = role;
    logActivity(`Updated account: @${uname}`);
  } else {
    if (!pass) { notify('Password required for new account', 'error'); return; }
    db.accounts.push({ id: uid(), username: uname, display: disp, password: pass, role, created: Date.now() });
    logActivity(`Created account: @${uname} (${role})`);
  }

  saveDB(db);
  renderAccounts();
  closeModal('modal-account');
  notify('Account saved!', 'success');

  // Update topbar if editing self
  if (id === session.id) {
    session = db.accounts.find(a => a.id === session.id);
    document.getElementById('topbar-username').textContent = session.display || session.username;
  }
}

function confirmDeleteAccount(id) {
  const a = db.accounts.find(x => x.id === id);
  openConfirm(
    `Delete account @${a.username}?`,
    `This will permanently delete the account "${a.display || a.username}". Posts will remain.`,
    () => {
      db.accounts = db.accounts.filter(x => x.id !== id);
      saveDB(db);
      renderAccounts();
      logActivity(`Deleted account: @${a.username}`);
      notify('Account deleted', 'success');
    }
  );
}

function downloadExport() {
  const val = document.getElementById('export-post-select').value;
  let filename, content;
  if (val === '__index__') {
    filename = 'index.html';
    content = generateIndexHTML();
  } else {
    const p = db.posts.find(x => x.id === val);
    if (!p) return;
    filename = (p.slug || p.id) + '.html';
    content = generatePostHTML(p);
  }
  const blob = new Blob([content], { type: 'text/html' });
  const a = document.createElement('a');
  a.href = URL.createObjectURL(blob);
  a.download = filename;
  a.click();
  notify(`Downloaded ${filename}`, 'success');
}

function uid() { return Math.random().toString(36).slice(2, 10) + Date.now().toString(36); }

function esc(str) {
  return String(str || '').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function slugify(str) {
  return str.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-|-$/g, '');
}

function todayStr() {
  return new Date().toISOString().split('T')[0];
}

function fmtSize(bytes) {
  if (!bytes) return 'â€”';
  if (bytes < 1024) return bytes + ' B';
  if (bytes < 1048576) return (bytes/1024).toFixed(1) + ' KB';
  return (bytes/1048576).toFixed(1) + ' MB';
}

function timeAgo(ts) {
  if (!ts) return 'â€”';
  const diff = Date.now() - ts;
  const m = Math.floor(diff / 60000);
  if (m < 1)  return 'just now';
  if (m < 60) return `${m}m ago`;
  const h = Math.floor(m / 60);
  if (h < 24) return `${h}h ago`;
  return `${Math.floor(h/24)}d ago`;
}
